## Installation

```shell
	$ npm install chainchainchain
```


## Usage

```javascript
const chain = require('chainchainchain')

let obj = chain({x:'chain'}, {y:'chain'}, {z:'chain'})

obj.x // 'chain'
obj.y // 'chain'
obj.z // 'chain'
```
